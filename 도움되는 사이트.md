advance는 iterator의 위치를 n만큼 이동시켜서 그때의 값이 출력이 가능하다.

    std::vector<int>::iterator it = numbers.begin(); // 반복자를 컨테이너의 첫 번째 원소로 초기화

    // it 반복자를 2번 앞으로 이동
    std::advance(it, 2);
    
    // 이동된 위치에서 원소 출력
    std::cout << "Value at advanced position: " << *it << std::endl;
